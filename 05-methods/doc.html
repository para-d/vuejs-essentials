Methods

Methods can be declared in the options array.

Declaration

            let app = Vue.createApp({
                methods: {
                    method1: function(){
                        ...
                    },
                    method2() {
                        ...
                    }
                }
            });

Accessing Data Variables

            let app = Vue.createApp({
                data: function(){
                    return {
                        var1: 'some text',
                        var2: 3
                    };
                },
                methods: {
                    method1: function(){
                        console.log(this.var1);
                    },
                    method2() {
                        this.var2+=1;
                    }
                }
            });

Calling

Just by the name

            <div id="app" v-cloak>
                <div class="box">
                    {{logData}}
                </div>
                <button @click="method1">Click to call method1 [name only]</button>

            </div>
            <script type="text/javascript">
                let app = Vue.createApp({
                    data: function(){
                        return {
                            logData: ''
                        };
                    },
                    methods: {
                        method1: function(){
                            this.logData = 'method1 called';
                        }
                    }
                });

                app.mount('#app');
            </script>

With parenthesis

            <div id="app" v-cloak>
                <div class="box">
                    {{logData}}
                </div>
                <button @click="method1()">Click here</button>

            </div>
            <script type="text/javascript">
                let app = Vue.createApp({
                    data: function(){
                        return {
                            logData: ''
                        };
                    },
                    methods: {
                        method1: function(){
                            this.logData = 'method1 called with parenthesis';
                        }
                    }
                });

                app.mount('#app');
            </script>

With Parameters

            <div id="app" v-cloak>
                <div class="box">
                    {{logData}}
                </div>
                <button @click="method1('text1', 'text2')">Click here</button>

            </div>
            <script type="text/javascript">
                let app = Vue.createApp({
                    data: function(){
                        return {
                            logData: ''
                        };
                    },
                    methods: {
                        method1: function(param1,param2){
                            this.logData = 'method1 called with param1: '+param1+' param2: '+param2;
                        }
                    }
                });

                app.mount('#app');
            </script>

With Events and Parmeters

There are two ways of passing an event to a method.

  1. Passing directly using $event variable
        <div id="app" v-cloak>
        <div class="box">
            {{logData}}
        </div>
        <button @click="method1('text1',$event)">Click here</button>

    </div>
    <script type="text/javascript">
        let app = Vue.createApp({
            data: function(){
                return {
                    logData: ''
                };
            },
            methods: {
                method1: function(param1,event){
                    this.logData = 'method1 called with param1: '+param1+' through '+event.type+' event';
                }
            }
        });

        app.mount('#app');
    </script>
  2. Passing as a parameter to an inline function
        <div id="app" v-cloak>
        <div class="box">
            {{logData}}
        </div>
        <button @click="(event) => method1('text2',event)">Click here</button>

    </div>
    <script type="text/javascript">
        let app = Vue.createApp({
            data: function(){
                return {
                    logData: ''
                };
            },
            methods: {
                method1: function(param1,event){
                    this.logData = 'method1 called with param1: '+param1+' through '+event.type+' event';
                }
            }
        });

        app.mount('#app');
    </script>
